constants {
    // Factors
    DECA  [da]      : 10^1 
    HECTO [h]       : 10^2
    PICO  [p]       : 10^-12
    FEMTO [f]       : 10^(-15)
    
    // Other constants
    AVOGRADO [A]    : 6.022 * 10^23
    KELVINFACTOR    : (263.15 + 10)
}

units {
    // Basic Types (default is numeric)
    distance  [m]
    time      [s]
    absTemp   [K]   
    energy    [J]  
    
    // Derived Types
    power     [W]   : energy / time
    temp      [C]   : KELVINFACTOR * absTemp
    velocity        : distance / time
}
