units {
	
	meter "m";
	doubleMeter "dm";
	tripleMeter "tm";
	
	bar "b";
    
	distance [meter]	: (1/2)doubleMeter | (1/2.5)bar;
    
	area "m^2"		: meter*meter;
	volume "m^3"	: area*meter;
	crazyVolume "mdmtm" : meter*doubleMeter*tripleMeter;
    
	kilo		"kg";
	doubleKilo	"dkg";
    
	weight [kilo]		: (1)kilo | (1/2)doubleKilo;
	
}
