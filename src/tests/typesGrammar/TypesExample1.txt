types {
    // Basic Types (default is numeric)
    meter	  "m"
    inch	  "in"
    time      "s"
    absTemp   "K"   
    energy    "J"  
    
    pessoa    "p"
    
    
    // Derived Types
    distance  		  "m"   : (1) meter | (2) inch
    power     		  "W"   : energy / time
    temp      		  "C"   : 0.5 * absTemp
    velocity          "m/s"	: distance / time
    
    animais 		  "a" 	: (1) pessoa | (1) vaca
	animaisTerrestres "at"	: (2) pessoa | (4) vaca
	
}

prefixes {
    DECA  "da"      : 10^1 
    HECTO "h"       : 10^2
    PICO  "p"       : 10^-12
    FEMTO "f"       : 10^(-15)
}
