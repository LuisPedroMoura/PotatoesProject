// Types Grammar Example 3
// This example will generate an error for the last type (mile)
// Inês Justo (84804), Luis Pedro Moura (83808), Maria João Lavoura (84681), Pedro Teixeira (84715)

prefixes {
    DECA  "da"      : 10^1 
    HECTO "h"       : 10^2
    PICO  "p"       : 10^-12
    FEMTO "f"       : 10^(-15)
}

types {
	// Basic Types (default is numeric)
	meter	 "m"
	inch 	 "inch"
	yard 	 "yd"
	distance "d"	: (-1+ 49725)meter 		 | (0.025)inch | (0.9)yard
	mile 	 "mil" 	: (0.75)distance | meter		// should give error
}
