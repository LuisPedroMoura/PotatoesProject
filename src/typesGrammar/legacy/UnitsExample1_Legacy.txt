DECA  [da]      : factor 10^1
HECTO [h]       : 10^2
PICO  [p]       : 10^-12
HELLLOOOOOOFEMTO [f]       : 10^(-15)
TEN             : 10

// Basic Types
distance  [m]   : numeric
time      [s]   : numeric
abs_temp  [K]   : numeric
energy    [J]   : numeric

// Derived Types
power     [W]   : energy / time
temp      [C]   : (263.15 + 10) * abs_temp
temp1     [CC]  : (263.15 + TEN) * abs_temp - time
velocity        : distance / time
